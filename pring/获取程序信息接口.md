# 获取程序信息接口

参考Spring Boot Actuator

```java
@Controller
@RequestMapping("/appInfo")
public class AppInfoController {
    private static final Logger logger = LoggerFactory.getLogger(AppInfoController.class);

    private static final SimpleDateFormat yyyyMMddHHmm = new SimpleDateFormat("yyyy-MM-dd HH:mm");

    private static final BigDecimal gb = new BigDecimal(1024 * 1024 * 1024);

    @Value("${build-time}")
    private String buildTime;

    @Autowired
    private DataSource dataSource;
    @Value("${spring.datasource.url}")
    private String spring_datasource_url;

    @GetMapping
    @ResponseBody
    public ResponseEntity<String> state() {
        Map<String, String> infoMap = new LinkedHashMap<>();

        String pidAndDeviceName = ManagementFactory.getRuntimeMXBean().getName();
        if (StringUtils.hasText(pidAndDeviceName)) {
            String[] temp = pidAndDeviceName.split("@");
            infoMap.put("pid", temp[0]);
            if (temp.length > 1) infoMap.put("设备名称", temp[1]);
        }

        // 获取每个磁盘的使用量
//        FileSystems.getDefault().getRootDirectories().forEach(root -> {
//            try {
//                FileStore fileStore = Files.getFileStore(root);
//                logger.info("盘符： {}", root);
//                logger.info("磁盘名称： {}", fileStore.name());
//                logger.info("类型： {}", fileStore.type());
//                logger.info("只读： {}", fileStore.isReadOnly());
//                logger.info("总量： {}", fileStore.getTotalSpace());
//                logger.info("可用： {}", fileStore.getUsableSpace());
//            } catch (IOException e) {
//                e.printStackTrace();
//            }
//        });
        try {
            Path path = new File(".").toPath();
            FileStore fileStore = Files.getFileStore(path);
            infoMap.put("磁盘类型", fileStore.type());
            infoMap.put("磁盘只读", fileStore.isReadOnly() ? "是" : "否");
            infoMap.put("磁盘总容量", BigDecimal.valueOf(fileStore.getTotalSpace()).divide(gb).setScale(2, RoundingMode.HALF_UP) + " GB");
            infoMap.put("磁盘可用容量", BigDecimal.valueOf(fileStore.getUsableSpace()).divide(gb).setScale(2, RoundingMode.HALF_UP) + " GB");
        } catch (IOException e) {
            e.printStackTrace();
        }

        // OperatingSystemMXBean位于com.sun.management中，某些系统(例如Android)或java版本可能不可用
        OperatingSystemMXBean os = (OperatingSystemMXBean) ManagementFactory.getOperatingSystemMXBean();
        infoMap.put("内存总容量", BigDecimal.valueOf(os.getTotalPhysicalMemorySize()).divide(gb).setScale(2, RoundingMode.HALF_UP) + " GB");
        infoMap.put("内存可用容量", BigDecimal.valueOf(os.getFreePhysicalMemorySize()).divide(gb).setScale(2, RoundingMode.HALF_UP) + " GB");
        infoMap.put("cpu使用率", BigDecimal.valueOf(os.getSystemCpuLoad() * 100).setScale(2, RoundingMode.HALF_UP) + " %");

        infoMap.put("项目位置", new File("").getAbsolutePath());
        // 配合extool-maven-plugin使用
        infoMap.put("构建时间", buildTime);

        // 参考spring boot actuator的DataSourceHealthIndicator
        infoMap.put("数据库URL", spring_datasource_url);
        JdbcTemplate jdbcTemplate = new JdbcTemplate(dataSource);
        Boolean isDbValid = jdbcTemplate.execute((ConnectionCallback<Boolean>) connection ->  {
            infoMap.put("数据库名称", connection.getMetaData().getDatabaseProductName());
            infoMap.put("数据库版本", connection.getMetaData().getDatabaseProductVersion());
            return connection.isValid(0);
        });
        infoMap.put("数据库状态", Optional.ofNullable(isDbValid).orElse(false) ? "可用" : "不可用");

        infoMap.put("服务器时间", yyyyMMddHHmm.format(new Date()));

        StringBuilder infoStr = new StringBuilder();
        infoMap.forEach((k, v) -> {
            infoStr.append(k).append(": ").append(v).append("\n");
        });

        HttpHeaders responseHeaders = new HttpHeaders();
        responseHeaders.set(HttpHeaders.CONTENT_TYPE, "text/plain;charset=UTF-8");
        return ResponseEntity.ok()
                .headers(responseHeaders)
                .body(infoStr.toString());
    }
}
```
