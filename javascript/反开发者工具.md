# 反开发者工具

```javascript
/**
 * 反开发者工具
 * @param timeout 循环检测时间(毫秒，默认500)
 */
function antiDevtools(timeout) {
    let div = document.createElement('div');
    // 当调用了div的id属性的getter方法时
    Object.defineProperty(div, 'id', {
        get: () => {
            // webpack打包时会清除debugger语句，间接执行debugger
            // eslint-disable-next-line no-eval
            eval('debugger;');
        }
    });
    // 轮询向控制台输出div元素，
    // 如果用户打开了开发者工具，会一直进入debugger状态
    setInterval(() => {
        console.log(div);
        console.clear();
    }, timeout || 500);
}
```
